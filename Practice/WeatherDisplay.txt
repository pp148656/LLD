public class WeatherData{
    private int temperature;
    private int humidity;
    private int windSpeed;
    //public getters and setters
}
public class WeatherStation{
        List<WeatherSubscriber> weatherSubscriber;
        private WeatherData weatherData;
        public WeatherStation (WeatherData weatherData){
              this.weatherData= weatherData;
        }
        public void setWeatherData(WeatherData weatherData){
            this.weatherData= weatherData;
            notifySubscribers(weatherData);
        }
        private void notifySubscribers(){
            for(subscriber:weatherSubscriber){
                subscriber.update();
            }
        }
        public WeatherData getWeatherData(){
            return this.weatherData;
        }
        //keep separate getters for temp, humidity, wind speed

        public void addSubscriber(Subscriber sub){
            weatherSubscriber.add(sub);
        }
        public void removeSubscriber(Subscriber sub){
            weatherSubscriber.erase(sub);
        }
}
public interface Subscriber{
    public void update();
}
