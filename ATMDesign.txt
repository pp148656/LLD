public interface ATMState {
       public boolean readCard(CardDetails cardDetails, UUID tid);
       public boolean readWithdrawalDetails(CardDetails cardDetails,int amount ,UUID tid);
       public boolean ejectCard();
       public UUID init();
       public boolean cancelTransaction();
}
public ReadyState implements ATMState{
     private final ATMMachine atmMachine;

     //a constructor that accepts ATMMachine
     @Override
     public UUID init(){
       UUID tid= UUID.random();
       //save in db
       //its very important to pass this as all states should have same machine ref
       this.atmMachine.changeState(new CardReadingState(this));
       return tid;
     }
     //others overrides throw exception here

}
public CardReadingState implements ATMState{
     private final ATMMachine atmMachine;

     //a constructor that accepts ATMMachine
     @Override
     public boolean readCard(CardDetails cardDetails, UUID tid){
        //call and API to validate the card

        //persistInDB(tid, cardDetails);
        // if anything fails return false
        this.atmMachine.changeState(new WithdrawalDetailsReadingState(this));
        return true;
     }
     @Override
     public boolean cancelTransaction(UUID tid){
           this.atmMachine.changeState(new ReadyState(this));
           return true;
     }
     //others overrides throw exception here
}
